<%= display_categories(@collection.categories, nil) do |category|
     check_box_tag("#{article.id}_#{category.id}",
                   "1",
                   my_categories.include?(category),
                   { :id => "#{article.id}_#{category.id}" }) +
     observe_field("#{article.id}_#{category.id}",
                   { :url => url_for({ :controller => 'article',
                                       :action => 'article_category',
                                       :category_id => category.id,
                                       :article_id => article.id }),
                     :with => "'status='+element.checked"
                   }) +
     category.title
   end %>

<%= display_categories(@collection.categories, nil) do |category|
              category.title +
              " " +
              link_to('delete', 
                     { :action => 'delete',
                       :category_id => category.id,
                       :collection_id => @collection.id },
                     { :confirm => "Are you sure you want to delete #{category.title} and all its subcategories?"}) +
             " " +
              link_to_function('add child', 
                               "add_child_prompt('#{category.id}', '#{category.title}')", 
                               { :title => "Add a subcategory underneath #{category.title}"} )
   end %>
